class Solution {
public:
     
  void  help(TreeNode * root, vector<string> &res ,string path)
    {
      if(!root)
          return ;
        
        
        if(!root->left && !root->right)//leaf
        {
                   path =path+ to_string( root->val);
             res.push_back(path);
            path.clear();
            return;
        }
       path =path+ to_string( root->val)+"->";
        if(root->left)
            help(root->left , res , path );
        if(root->right)
              help(root->right ,res , path);
      return;
    }
    vector<string> binaryTreePaths(TreeNode* root) {
     string path=""; 
        vector<string> res;
        if(!root)
            return {};
        help(root , res ,path);
      return res;  
    }
};
